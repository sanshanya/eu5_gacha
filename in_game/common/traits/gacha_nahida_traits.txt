gacha_nahida_origin_trait = {
  category = ruler  # 注意：origin和awakened使用同一category，可以共存
  allow = {
    always = no
  }

  modifier = {
    global_institution_growth_modifier = 0.15
    cultural_influence_modifier       = 0.10
    legislative_efficiency            = 0.10
  }
}

gacha_nahida_awakened_trait = {
  category = ruler  # 与origin相同，不会冲突
  allow = {
    always = no
  }

  modifier = {
    embrace_institution_cost_modifier  = -0.10
    global_devastation_recovery        = 0.15
    global_estate_satisfaction_recovery = 0.10
  }
}

gacha_nahida_transcended_trait = {
  category = general  # ⚠️ 必须不同！否则会移除origin trait
  allow = {
    always = no
  }

  modifier = {
    army_auxiliary_power   = 0.10
    army_artillery_power   = 0.10
    land_morale_modifier   = 0.05
  }
}
