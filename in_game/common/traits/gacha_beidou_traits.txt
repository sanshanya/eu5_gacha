# Beidou Traits - V3 Design "海上女王"

# C0 初始三格（海上女王）

# 1. 南十字的旗号（光环）
gacha_beidou_origin_trait = {
    category = ruler
    allow = { always = no }
    # 海军士气/私掠效率（由国家修正提供）
    modifier = { gacha_country_aura = yes }
}

# 2. 斩浪者（职业·海军）
gacha_beidou_awakened_trait = {
    category = general
    allow = { always = no }
    # 海军上限/耐久的替代表现
    modifier = {
        ship_build_speed = 0.10
        naval_damage_taken = -0.10
    }
}

# 3. 不羁的船长（统治者）
gacha_beidou_transcended_trait = {
    category = ruler
    allow = { always = no }
    # 降低附庸不满与叛乱压力（替代“严苛镇压花费”）
    modifier = {
        subject_loyalty = 0.10
        monthly_rebel_growth = -0.10
    }
}

# 命座进阶三格（从“客将”到“兄弟”）

# C2：怒海横行（光环升级）
gacha_beidou_c2_trait = {
    category = ruler
    allow = { always = no }
    modifier = { gacha_country_aura = yes }
}

# C4：踏海斩鲸（职业升级）
gacha_beidou_c4_trait = {
    category = general
    allow = { always = no }
    modifier = {
        ship_build_speed = 0.20
        naval_damage_taken = -0.20
        naval_unit_attrition = -0.25
    }
}

# C6：海上霸主（统治者升级）
gacha_beidou_c6_trait = {
    category = ruler
    allow = { always = no }

    }
