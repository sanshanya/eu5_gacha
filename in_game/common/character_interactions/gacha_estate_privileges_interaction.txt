gacha_estate_privileges_interaction = {
  message = no
  is_consort_action = no
  on_own_nation = yes
  ai_tick = never
  context_menu_click_mode = click

  potential = {
    scope:actor = { has_ruler = yes }
  }

  allow = {
    always = yes
  }

  select_trigger = {
    looking_for_a = character
    source = actor
    target_flag = recipient
    name = "gacha_select_1title"
    column = { data = name }
    visible = {
      is_alive = yes
      is_ruler = yes
    }
  }

  effect = {
    scope:actor = {
      # 为天外之人解锁权限并挂载国家修正
      add_country_modifier = { modifier = gacha_estate_integration_modifier years = -1 mode = add_and_extend }

      # 给首都位置添加少量天外之人 POP，确保阶层有可见权力
      random_owned_location = {
        limit = { is_capital = yes }
        add_pop = {
          type = pop_type:gacha
          size = 100
          culture = owner.dominant_culture
          religion = owner.religion
        }
      }
    }
  }

  ai_will_do = { value = 0 }
}
