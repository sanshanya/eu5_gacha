on_actions = {
    on_game_start = {
        effect = {
            # 提前为玩家国家初始化抽卡相关变量，避免首次点击祈愿时报“变量未设置”的错误
            every_country = {
                limit = { is_ai = no }
                gacha_ensure_state_initialized = yes
                add_country_modifier = { modifier = gacha_estate_integration_modifier years = -1 mode = add_and_extend }
                # 默认解锁天外之人可入内阁/领兵的特权，避免被阻挡
                add_estate_privilege = { estate = estate_type:gacha_estate privilege = estate_privilege:gacha_in_administration }
                add_estate_privilege = { estate = estate_type:gacha_estate privilege = estate_privilege:gacha_command_positions }
            }
        }
    }
    on_game_loaded = {
        effect = {
            every_country = {
                limit = { is_ai = no }
                gacha_ensure_state_initialized = yes
                add_country_modifier = { modifier = gacha_estate_integration_modifier years = -1 mode = add_and_extend }
                add_estate_privilege = { estate = estate_type:gacha_estate privilege = estate_privilege:gacha_in_administration }
                add_estate_privilege = { estate = estate_type:gacha_estate privilege = estate_privilege:gacha_command_positions }
            }
        }
    }
}
